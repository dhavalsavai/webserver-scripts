server {
    listen 443 ssl;
    server_name example.com www.example.com;

    ssl_certificate /home/ubuntu/cert/techin.crt;
    ssl_certificate_key /home/ubuntu/cert/techin.key;

    location / {
        proxy_pass https://www.example.com:445/;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;

        proxy_ssl_verify off;
        proxy_ssl_server_name on;
    }

    error_log /var/log/nginx/mainsite-error.log;
    access_log /var/log/nginx/mainsite-access.log;
}
